%% ---- SLIDE 1: Motivation ----
\begin{frame}{Parental Investments and Misinformation}

\textbf{Parental effort} is a key input in children's human capital --- but optimal investment requires accurate beliefs about the child's performance.

\bigskip
\textbf{Parents are often misinformed:}
\begin{itemize}
    \item Belief--performance gap $>1$ SD; 1/3 wrong about which child does better \\ {\small (Dizon-Ross, 2019 --- Malawi)}
    \item Parents vastly underestimate missed assignments {\small (Bergman, 2021 --- LA)}
    \item Misperceptions larger among low-SES households
\end{itemize}

\bigskip
\textbf{The puzzle:} SMS with the \emph{same} information as report cards $\Rightarrow$ GPA $+0.19$ SD, failures $-38\%$, at $<$\$63/student.

\medskip
Report cards are free. \textbf{Why don't parents use them?}

\end{frame}

%% ---- SLIDE 2: What Could Explain This? ----
\begin{frame}{What Could Explain This?}

\textbf{Non-behavioral:}
\begin{itemize}
    \item \textbf{Moral hazard:} children hide report cards --- but 59\% of Bergman's effect is belief revision, not monitoring
    \item \textbf{Constraints:} hard to reconcile with large effects on parental time/effort
\end{itemize}

\medskip
\textbf{Behavioral --- two candidates, hard to separate:}
\begin{itemize}
    \item \textbf{Asymmetric belief updating:} parents discount bad signals (motivated reasoning); SMS forces engagement with the signal
    \item \textbf{Salience of the channel:} SMS creates contrast with the prior that ``all is fine'' (Bordalo et al., 2012); report card is cognitively invisible under load (Gabaix, 2014)
    \item[] $\Rightarrow$ Both predict: same content, new channel $=$ better outcomes
\end{itemize}

\medskip
\textbf{Open questions:}
\begin{itemize}
    \item Can we separate salience from belief updating with a single RCT?
    \item Does the effect persist, or does salience decay?
\end{itemize}

\end{frame}
